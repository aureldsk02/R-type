163 void animate(double delta) {
164 this->frame_timer += delta;
165
166 if (this->frame_timer >= this->frame_duration) {
167 this->frame_timer = 0;
168 this->frame_rect.left += this->frame_rect.width;
169 if (this->frame_rect.left >=
170 this->sprite.getTexture()->getSize().x - 5) {
171 this->frame_rect.left = 0;
172 done_once = true;
173 }
174 this->sprite.setTextureRect(this->frame_rect);
175 }
176 }
