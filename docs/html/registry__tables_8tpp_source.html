35template<typename Component>
42vector<optional<Component>> &registry::register_components()
43{
44 // get the component type
45 type_index new_type = typeid(Component);
46
47 // Check if the corresponding type table is already stored or not
48 if (_components_arrays.find(new_type) == _components_arrays.end()) {
49 // If not found, create the component table
50 _components_arrays.emplace(
51 new_type,
52 std::make_any<std::vector<std::optional<Component>>>()
53 );
54 }
55
56 // Get the final table
57 auto &table = std::any_cast<std::vector<std::optional<Component>>&>(_components_arrays[new_type]);
58
59 // Create a lambda that will erase this type of component
60 // for a given entity
61 auto delete_func = [&table] (const entity &e) {
62 if (e.getId() < table.size())
63 table[e].reset();
64 };
65
66 // Add the lambda to the erasing functions container
67 _erase_functions[new_type] = delete_func;
68
69 // Return the reference to the final table (newly created or not)
70 return table;
71}
